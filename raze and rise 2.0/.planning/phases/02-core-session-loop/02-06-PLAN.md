       ```typescript
       export interface SetRowState { setNumber: number; result: 'go' | 'no-go' | null }

       export function supersetRoundComplete(
         setsA: SetRowState[],
         setsB: SetRowState[],
         setNumber: number,
       ): boolean {
         const a = setsA.find(s => s.setNumber === setNumber);
         const b = setsB.find(s => s.setNumber === setNumber);
         return Boolean(a?.result && b?.result);
       }

       export interface SupersetTarget {
         exerciseLetter: 'A' | 'B';
         setNumber: number;
       }

       export function nextSupersetTarget(
         currentArm: 'A' | 'B',
         currentSetNumber: number,
         setsA: SetRowState[],
         setsB: SetRowState[],
         maxSets: number,
       ): SupersetTarget | null {
         if (currentArm === 'A') {
           return { exerciseLetter: 'B', setNumber: currentSetNumber };
         }
         // currentArm === 'B'
         if (!supersetRoundComplete(setsA, setsB, currentSetNumber)) {
           return { exerciseLetter: 'A', setNumber: currentSetNumber };  // wait for A
         }
         if (currentSetNumber >= maxSets) return null;
         return { exerciseLetter: 'A', setNumber: currentSetNumber + 1 };
       }

       // FlashList data array resolution
       export type FlashListItem =
         | { type: 'single'; exerciseId: string }
         | { type: 'superset-pair'; exerciseAId: string; exerciseBId: string; groupId: number };

       export function buildFlashListData(
         exercises: Array<{ id: string; supersetGroup: number | null }>,
       ): FlashListItem[] {
         const out: FlashListItem[] = [];
         const seenGroups = new Set<number>();
         for (let i = 0; i < exercises.length; i++) {
           const ex = exercises[i];
           if (ex.supersetGroup == null) {
             out.push({ type: 'single', exerciseId: ex.id });
           } else if (!seenGroups.has(ex.supersetGroup)) {
             const partner = exercises.find((e, j) => j > i && e.supersetGroup === ex.supersetGroup);
             if (partner) {
               out.push({ type: 'superset-pair', exerciseAId: ex.id, exerciseBId: partner.id, groupId: ex.supersetGroup });
               seenGroups.add(ex.supersetGroup);
             } else {
               // Orphan — render as single
               out.push({ type: 'single', exerciseId: ex.id });
             }
           }
           // else: partner already emitted via the seenGroups check
         }
         return out;
       }

       export function findFlashListIndexForExercise(
         data: FlashListItem[],
         exerciseId: string,
       ): number {
         return data.findIndex(item =>
           item.type === 'single'
             ? item.exerciseId === exerciseId
             : item.exerciseAId === exerciseId || item.exerciseBId === exerciseId
         );
       }
       ```
