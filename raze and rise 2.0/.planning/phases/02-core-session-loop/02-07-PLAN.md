    Props: `{ visible: boolean; onFadeOutComplete: () => void }`. The parent (SessionScreen) controls visibility and is responsible for navigating after `onFadeOutComplete`.

    Structure:
    ```typescript
    import { useEffect } from 'react';
    import { View } from 'react-native';
    import Animated, { useSharedValue, useAnimatedStyle, withTiming, useReducedMotion, runOnJS } from 'react-native-reanimated';
    import LottieView from 'lottie-react-native';

    export function AnubisOverlay({ visible, onFadeOutComplete }: AnubisOverlayProps) {
      const opacity = useSharedValue(0);
      const reducedMotion = useReducedMotion();
      const lottieRef = useRef<LottieView>(null);

      // Fade in on mount/visible
      useEffect(() => {
        if (!visible) return;
        opacity.value = reducedMotion ? 1 : withTiming(1, { duration: 600 });
        if (reducedMotion) {
          // Skip animation; hold first frame for 600ms then fade out
          const t = setTimeout(handleLottieDone, 600);
          return () => clearTimeout(t);
        }
      }, [visible, reducedMotion]);

      const handleLottieDone = () => {
        opacity.value = reducedMotion
          ? 0
          : withTiming(0, { duration: 600 }, (finished) => {
              if (finished) runOnJS(onFadeOutComplete)();
            });
        if (reducedMotion) {
          // No animation callback path — fire immediately
          onFadeOutComplete();
        }
      };

      const animatedStyle = useAnimatedStyle(() => ({ opacity: opacity.value }));

      if (!visible) return null;

      return (
        <Animated.View
          pointerEvents="auto"
          accessibilityRole="image"
          accessibilityLabel="Workout complete"
          style={[
            { position: 'absolute', top: 0, left: 0, right: 0, bottom: 0, backgroundColor: '#0A0A0B', justifyContent: 'center', alignItems: 'center' },
            animatedStyle,
          ]}
        >
          <LottieView
            ref={lottieRef}
            source={require('../../../assets/animations/anubis.json')}
            autoPlay={!reducedMotion}
            loop={false}
            style={{ width: '60%', height: '60%' }}
            onAnimationFinish={handleLottieDone}
          />
        </Animated.View>
      );
    }
    ```

    Key behaviors per UI-SPEC.md:
    - Fade-in 600ms `anubis-fade` ease-in-out
    - Lottie plays once (autoPlay, loop=false)
    - onAnimationFinish → fade-out 600ms → onFadeOutComplete callback
    - useReducedMotion: skip Lottie playback; hold first frame for 600ms; immediate transition
    - `accessibilityRole="image"` + `accessibilityLabel="Workout complete"` (UI-SPEC.md VoiceOver)
    - Full-screen `#0A0A0B` background (bg token hex — required for animated overlay style prop)
    - pointerEvents="auto" — blocks taps during playback (UI-SPEC.md "NOT dismissible — plays through to completion")

    Note: bg color is hard-coded hex `#0A0A0B` per Phase 1 + Phase 2 established exception for Animated.View style.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| writeTransaction multi-row commit | Failure mid-transaction would leave sessions in inconsistent state (session row written but rotation pointer not advanced, or vice versa) — PowerSync writeTransaction guarantees atomicity |
| MMKV cleanup ordering | If cleanup runs before transaction commits and transaction throws, the next session would start without crash recovery state |
| Anubis asset → bundled app | Same T-02-02 mitigation gate from Plan 01 — re-verified that no expressions field exists |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Information Disclosure | completeSession writes sessions row | mitigate | Phase 1 RLS on sessions table: `user_id = auth.uid()`. completeSession passes the authenticated user_id from useSession; PowerSync sync upload validates against Supabase RLS. |
| T-02-02 | Elevation of Privilege | anubis.json with expressions field | mitigate (carry from Plan 01) | Plan 01 Task 4 grep gate already ran. Plan 07 does not re-verify; relies on Plan 01's gate. |
| T-02-10 | Tampering | rotation pointer skipped due to partial commit | mitigate | writeTransaction atomicity. Test in Task 1 asserts both execute calls happen inside the transaction callback. |
| T-02-11 | Denial of Service | Android back-button bypass | mitigate | BackHandler listener returns true to consume the event; Alert.alert is the only path; default OS behavior is suppressed. |
</threat_model>
